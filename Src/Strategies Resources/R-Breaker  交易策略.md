def onTick():
    try:
        records =exchange.GetRecords()  #由你自己设定的周期数返回的数据
        HH = records[-2]['High'] #最日最高
        LC = records[-2]['Low']  #昨日最低

        R3 = HH +2*(Pivot-LC) #阻力3
